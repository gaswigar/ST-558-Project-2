lowerFn <- function(data, mapping, method = "lm", ...) {
  p <- ggplot(data = data, mapping = mapping) +
    geom_point()+
    geom_smooth(method="loess",color = "red", ...)+
    geom_smooth(method='lm', color = "blue", ...)
  p
}
